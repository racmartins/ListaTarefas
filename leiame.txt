drop table tasks;
drop table users;

CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(255),
  email VARCHAR(255)
);

CREATE TABLE tasks (
  id INT PRIMARY KEY AUTO_INCREMENT,
  task VARCHAR(255) NOT NULL,
  status VARCHAR(50),
  user_id INT NOT NULL,
  FOREIGN KEY (user_id) REFERENCES users(id)
);
